// FRAGMENTS
fragment Book on Book {
  id
  title
  description
  rating
}

fragment Review on Review {
  id
  comment
  rating
  title
  user {
    name
    imageUrl
  }
}

fragment SerchBook on SearchBookResult {
      id
    title
    description
    authors {
      name
    } 
    imageUrl
}

// QUERIES
HomePage query returns the list of the books and reviews
variables - {
  "size": "LARGE",
  "orderBy": "ID_DESC",
  "smallSize": "SMALL"
}
query HomePage(
  $size: ImageSize
  $smallSize: ImageSize
  $orderBy: BooksOrderBy
  $orderReviews: ReviewsOrderBy
) {
  books(orderBy: $orderBy) {
    ...Book
    imageUrl(size: $size)
    authors {
      name
    }
  }
  reviews(orderReviews: $orderReviews) {
    ...Review
    book {
      ...Book
      imageUrl(size: $smallSize)
    }
  }
}

HomePage query returns the particular book's data
query BookPage {
  book(bookId: "2") {
    ...Book
    imageUrl
    reviews {
      ...Review
    }
    authors {
      name
    }
  }
}

 }

  searchBook (query: $query) {
